import data from './data.js'

function getServerName(list, id) {
  let serverName
  ;(function getName(list, id) {
    for (let i=0; i<list.length; i++) {
      let children = list[i].children
      if (list[i].id === id) {
        serverName = list[i].name
      } else {
        if (!serverName) {
          children && getName(children, id)
        }
      }
    }
  })(list, id)
}

import data from './data.js'

function getName (list, id) {
	return list.reduce((pre, cur) => {
		if (cur.id === id) {
			return pre + cur.name
		} else if (cur.children) {
			return pre + getName(cur.children, id)
		}
	}, '')
}

getName(data, 122)
