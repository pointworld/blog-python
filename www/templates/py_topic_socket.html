整个计算机网络都是由协议组成的 


弄懂 HTTP、Socket、TCP 这几个概念






socket 并不是协议，它是一个接口，建立在 TCP 和 IP 之上，用于实现自己的应用层协议，实现网络编程


client 和 server 实现通信




server 端编程

# server 端编程

import socket


# socket.AF_INET: AddressFamily 指使用 ipv4 网络传输数据 -- 网络层协议
# socket.SOCK_STREAM: SocketKind 指使用流式数据 TCP -- 传输层协议
server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

server.bind(('0.0.0.0', 8111))

server.listen()

# sock 是 server 端和 client 端共同建立的 socket
# addr 是 client 端的地址，如 ('127.0.0.1', 63714)
sock, addr = server.accept()

# 获取从客户端发送的数据
# 一次获取 1k 的数据
data  = sock.recv(1024)

print(data.decode('utf8'))

sock.send('hello {}'.format(data.decode('utf8')).encode('utf8'))

server.close()
sock.close()

import socket

# server 端编程

import socket


server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind(('0.0.0.0', 8113))
server.listen()

sock, addr = server.accept()

while True:
    data  = sock.recv(1024)
    print(data.decode('utf8'))
    re_data = input()
    sock.send(re_data.encode('utf8'))

多对一聊天

# server 端编程

import socket
import threading


server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind(('0.0.0.0', 8115))
server.listen()

def handle_sock(sock, addr):
    while True:
        data  = sock.recv(1024)
        print(data.decode('utf8'))
        re_data = input()
        sock.send(re_data.encode('utf8'))

while True:
    sock, addr = server.accept()

    # 用线程去处理新接收的连接（用户）
    client_thread = threading.Thread(target=handle_sock, args=(sock, addr))
    client_thread.start()



# client 端编程

import socket


client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
client.connect(('127.0.0.1', 8115))

while True:
    re_data = input()
    client.send(re_data.encode('utf8'))
    data = client.recv(1024)
    print(data.decode('utf8'))


socket 模拟 HTTP 请求

# requests -> urllib -> socket

import socket
from urllib.parse import urlparse

def get_url(url):
    # 通过 socket 请求 html
    url = urlparse(url)
    host = url.netloc
    path = url.path
    if path == '':
        path = '/'

    # 建立 socket 连接
    client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    client.connect((host, 80))

    client.send('GET {} HTTP/1.1\r\nHost:{}\r\nConnection:close\r\n\r\n'.format(path, host).encode('utf8'))

    data = b''
    while True:
        d = client.recv(1024)
        if d:
            data += d
        else:
            break

    data = data.decode('utf8')
    html_data = data.split('\r\n\r\n')[1]
    print(html_data)
    client.close()


if __name__ == '__main__':
    get_url('http://www.baidu.com')
