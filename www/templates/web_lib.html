IE6/7/8 不兼容 
封装 class 类名

1）初始版

window.onload = function () {
    // 兼容方法：取出限定范围内的所有标签
    // 利用遍历的方法，通过每一个标签的 className 属性来判断
    // 若相等就添加到一个空数组中
    // 遍历完后，返回该数组即可

    // 封装自己的 class 类名
    function getClass(classname) {

        // 先进行判断，如果浏览器支持，则直接返回原生写法
        if (document.getElementsByClassName) {
            return document.getElementsByClassName(classname);
        }

        // 兼容 IE6、7、8 的写法
        var arr = [];  // 用于存放满足条件的 node 数组
        var dom = document.getElementsByTagName("*");
        var len = dom.length;
        for(var i=0;i<len;i++){
            if(dom[i].className===classname){
                arr.push(dom[i]);
            }
        }
        return arr;
    }

    // 调用封装函数
    getClass("demo");
}

2）升级版

window.onload = function () {
    // 兼容方法：取出限定范围内的所有标签
    // 利用遍历的方法，通过每一个标签的 className 属性来判断
    // 若相等就添加到一个空数组中，解决一个标签多类名的问题
    // 遍历完后，返回该数组即可

    // 封装自己的 class 类名
    function getClass(classname) {

        // 先进行判断，如果浏览器支持，则直接返回原生写法
        if (document.getElementsByClassName) {
            return document.getElementsByClassName(classname);
        }

        // 兼容 IE6、7、8 的写法
        var arr = [];  // 用于存放满足条件的 node 数组
        var dom = document.getElementsByTagName("*");
        var dom_len = dom.length;
        for(var i=0;i<dom_len;i++){
            var txtarr = dom[i].className.split(" "); // 字符串转换为数组
            var txtarr_len = txtarr.length;
            for(var j=0;j<txtarr_len;j++){
                if(txtarr[j]===classname){
                    arr.push(dom[i]);
                }
            }
        }
        return arr;
    }

    // 调用封装函数
    getClass("demo");
}

window.onload = function () {
    function getClass(classname,id) {
        // 浏览器支持写法
        if(document.getElementsByClassName){
            if(id){
                var eleId=document.getElementById(id);
                return eleId.getElementsByClassName(classname);
            }
            else{
                return document.getElementsByClassName(classname);
            }
        }

        // IE6、7、8 不支持写法
        if(id){
            var eleId=document.getElementById(id);
            var dom=eleId.getElementsByTagName("*");

            var dom = document.getElementsByTagName("*");

function getStringByteLength(str){
    var len=0,c=0,str_len=str.length;
    for(var i=0;i<str_len;i++){
        c=str.charCodeAt(i);
        c>=0&&c<=127?len++:len+=2;
    }
    return len;
}
